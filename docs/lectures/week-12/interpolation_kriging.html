# calculate fit for each approach (MAPE)
def mape(est, obs):
    err = est-obs
    aerr = numpy.abs(err)
    den = obs + (obs == 0)
    paerr = aerr / den
    paerr *= 100
    return paerr.mean()
mape(county_h3.nn1_est, county_h3.inches)
mape(county_h3.nn5_est, county_h3.inches)
mape(county_h3.nn5id_est, county_h3.inches)
mape(county_h3.ok_est, county_h3.inches)
